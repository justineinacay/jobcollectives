    const php = this.toPhp(usd);

  clientMargin(c){

    const revPhp = c.rateUsd * this.rate();

    if(!revPhp) return null;

    const cost = this.totalVaCostPhp();

/* ──────────────────────────────────────────────────────────

   SECTION 9 — SLA ENGINE

────────────────────────────────────────────────────────── */

const SLA = {

  slaMs: CFG.slaHours * 3600 * 1000,

  check(task){

    if(!task.createdAt) return 'ok';

    const created = new Date(task.createdAt).getTime();

    const responded = task.firstResponseAt ? new Date(task.firstResponseAt).getTime() : Date.now();

    const elapsed = responded - created;

    if(elapsed > this.slaMs) return 'breach';

    if(elapsed > this.slaMs * 0.75) return 'warn';

    return 'ok';

  },

  elapsed(task){

    if(!task.createdAt) return 0;

    const created = new Date(task.createdAt).getTime();

    const end = task.firstResponseAt ? new Date(task.firstResponseAt).getTime() : Date.now();

    return (end - created) / 3600000; // hours

  },

  badge(task){

    const status = this.check(task);

    if(task.done||task.firstResponseAt){

      const h = this.elapsed(task).toFixed(1);

      return status==='breach'

        ? `<span class="sla-badge sla-breach">⚠ ${h}h</span>`

        : `<span class="sla-badge sla-ok">✓ ${h}h</span>`;

    }

    const h = ((Date.now()-new Date(task.createdAt).getTime())/3600000).toFixed(1);

    return status==='breach'

      ? `<span class="sla-badge sla-breach">BREACH ${h}h</span>`

      : status==='warn'

      ? `<span class="sla-badge sla-warn">${h}h</span>`

      : `<span class="sla-badge sla-ok">${h}h</span>`;

  },

  complianceRate(tasks){

    const tracked = tasks.filter(t=>t.createdAt);

    if(!tracked.length) return 100;

    const compliant = tracked.filter(t=>this.check(t)!=='breach');

    return Math.round((compliant.length/tracked.length)*100);

  },

  vaRate(vaUid){

    const vaTasks = APP.data.tasks.filter(t=>t.assigneeUid===vaUid&&!t.archived);

    return this.complianceRate(vaTasks);

  }

};

  APP.timer.start=Date.now()-APP.timer.elapsed*1000;

  APP.timer.iv=setInterval(()=>{

    APP.timer.elapsed=Math.floor((Date.now()-APP.timer.start)/1000);

    const el=document.getElementById('timer-display');

    if(el){

      const h=String(Math.floor(APP.timer.elapsed/3600)).padStart(2,'0');

      const m=String(Math.floor((APP.timer.elapsed%3600)/60)).padStart(2,'0');

      const s=String(APP.timer.elapsed%60).padStart(2,'0');

      el.textContent=`${h}:${m}:${s}`;

    }

  },1000);

  // Swap button visually without full re-render

  const btn=document.querySelector('.timer-btn-start');

  if(btn){btn.className='timer-btn-stop';btn.textContent='⏹ Stop & Log';btn.setAttribute('onclick','stopTimer()');}

}


function stopTimer(){

  clearInterval(APP.timer.iv); APP.timer.iv=null; APP.timer.running=false;

  toast('Timer stopped. Log your entry below.','info');

  // Re-render to show log form with pre-filled time

  renderCurrentPage();

}


function resetTimer(){

  clearInterval(APP.timer.iv); APP.timer.iv=null; APP.timer.running=false; APP.timer.elapsed=0;

  renderCurrentPage();

}


async function logTime(vaId){

  const desc=document.getElementById('timer-desc')?document.getElementById('timer-desc').value.trim():'';

  const manualEl=document.getElementById('timer-manual');

  const manual=manualEl?parseInt(manualEl.value)||0:0;

  const timerMins=Math.floor(APP.timer.elapsed/60);

  const minutes=manual||timerMins;

  if(!minutes||minutes<=0){toast('Run the timer or enter minutes manually.','error');return;}

  if(!desc){toast('Please describe what you worked on.','error');return;}

  const projEl=document.getElementById('timer-project');

  const projVal=projEl?projEl.value:'';

  const proj=projVal.split('|');

  const va=APP.data.team.find(v=>v.id===vaId);

  await addDoc('timelogs',{

    vaUid:vaId,vaName:va?.displayName||'',vaInitials:va?.initials||'',

    date:today(),description:desc,

    clientId:proj[0]||null,project:proj[1]||proj[0]||'',

    minutes,archived:false,

    _actAction:`logged ${Math.floor(minutes/60)}h ${minutes%60}m`,_actIcon:'⏱',_actColor:'navy',_actRecord:desc.slice(0,40)

  });

  resetTimer();

  toast(`Logged ${Math.floor(minutes/60)}h ${minutes%60}m — great work! 🎉`,'success');

}
